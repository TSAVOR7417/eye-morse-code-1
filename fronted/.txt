    function textToMorse(text) {
        return text.toUpperCase().split('').map(char => {
            if (char === ' ') return '/';
            return morseCode[char] || char;
        }).join(' ');
    }

    function morseToText(morse) {
        return morse.split(' ').map(code => {
            if (code === '/') return ' ';
            return reverseMorseCode[code] || code;
        }).join('');
    }

    function generateMorseAudio(morse) {
        if (!audioContext) {
            audioContext = new (window.AudioContext || window.webkitAudioContext)();
        }
        
        const dotDuration = 0.1;
        const dashDuration = 0.3;
        const spaceDuration = 0.1;
        const charSpaceDuration = 0.3;
        const wordSpaceDuration = 0.7;
        
        const sampleRate = audioContext.sampleRate;
        let totalDuration = 0;
        
        for (let i = 0; i < morse.length; i++) {
            const char = morse[i];
            if (char === '.') {
                totalDuration += dotDuration;
            } else if (char === '-') {
                totalDuration += dashDuration;
            } else if (char === ' ') {
                totalDuration += charSpaceDuration;
            } else if (char === '/') {
                totalDuration += wordSpaceDuration;
            }
            if (i < morse.length - 1) {
                totalDuration += spaceDuration;
            }
        }
        
        const bufferLength = Math.ceil(totalDuration * sampleRate);
        const buffer = audioContext.createBuffer(1, bufferLength, sampleRate);
        const channelData = buffer.getChannelData(0);
        
        let time = 0;
        let sampleIndex = 0;
        
        for (let i = 0; i < morse.length; i++) {
            const char = morse[i];
            let duration = 0;
            let frequency = 600;
            
            if (char === '.') {
                duration = dotDuration;
            } else if (char === '-') {
                duration = dashDuration;
            } else if (char === ' ') {
                duration = charSpaceDuration;
                frequency = 0;
            } else if (char === '/') {
                duration = wordSpaceDuration;
                frequency = 0;
            }
            
            const toneSamples = Math.floor(duration * sampleRate);
            for (let j = 0; j < toneSamples; j++) {
                if (sampleIndex < bufferLength) {
                    if (frequency > 0) {
                        channelData[sampleIndex] = Math.sin(2 * Math.PI * frequency * time) * 0.5;
                    } else {
                        channelData[sampleIndex] = 0;
                    }
                    sampleIndex++;
                    time += 1/sampleRate;
                }
            }
            
            if (i < morse.length - 1 && morse[i+1] !== ' ' && morse[i+1] !== '/') {
                const spaceSamples = Math.floor(spaceDuration * sampleRate);
                for (let j = 0; j < spaceSamples; j++) {
                    if (sampleIndex < bufferLength) {
                        channelData[sampleIndex] = 0;
                        sampleIndex++;
                        time += 1/sampleRate;
                    }
                }
            }
        }
        
        audioBuffer = buffer;
        totalTimeEl.textContent = formatTime(totalDuration);
        audioProgress.style.width = '0%';
        currentTimeEl.textContent = '0:00';
        
        playAudioBtn.innerHTML = '<i class="fas fa-play"></i>';
        isPlaying = false;
    }

    function playMorseAudio() {
        if (!audioBuffer) return;
        
        const source = audioContext.createBufferSource();
        source.buffer = audioBuffer;
        
        const gainNode = audioContext.createGain();
        gainNode.gain.value = 0.5;
        
        source.connect(gainNode);
        gainNode.connect(audioContext.destination);
        
        startTime = audioContext.currentTime;
        source.start(0);
        
        playAudioBtn.innerHTML = '<i class="fas fa-pause"></i>';
        isPlaying = true;
        
        progressInterval = setInterval(() => {
            const elapsed = audioContext.currentTime - startTime;
            const duration = audioBuffer.duration;
            
            if (elapsed >= duration) {
                stopAudio();
                return;
            }
            
            const progress = (elapsed / duration) * 100;
            audioProgress.style.width = `${progress}%`;
            currentTimeEl.textContent = formatTime(elapsed);
        }, 50);
        
        source.onended = function() {
            stopAudio();
        };
    }

    function playTextToSpeech(text) {
        if (!text || !('speechSynthesis' in window)) {
            alert('Text-to-speech not supported in this browser.');
            return;
        }
        
        // Cancel any ongoing speech
        if (speechUtterance) {
            speechSynthesis.cancel();
        }
        
        speechUtterance = new SpeechSynthesisUtterance(text);
        speechUtterance.rate = 0.9;
        speechUtterance.pitch = 1;
        speechUtterance.volume = 1;
        
        // Estimate duration based on text length
        const estimatedDuration = text.length * 0.2;
        totalTimeEl.textContent = formatTime(estimatedDuration);
        
        playAudioBtn.innerHTML = '<i class="fas fa-pause"></i>';
        isPlaying = true;
        
        let startTime = Date.now();
        progressInterval = setInterval(() => {
            const elapsed = (Date.now() - startTime) / 1000;
            const progress = Math.min((elapsed / estimatedDuration) * 100, 100);
            audioProgress.style.width = `${progress}%`;
            currentTimeEl.textContent = formatTime(elapsed);
            
            if (progress >= 100) {
                stopAudio();
            }
        }, 50);
        
        speechUtterance.onend = function() {
            stopAudio();
        };
        
        speechUtterance.onerror = function() {
            stopAudio();
        };
        
        speechSynthesis.speak(speechUtterance);
    }

    function stopAudio() {
        if (progressInterval) {
            clearInterval(progressInterval);
            progressInterval = null;
        }
        
        if (speechUtterance && 'speechSynthesis' in window) {
            speechSynthesis.cancel();
            speechUtterance = null;
        }
        
        playAudioBtn.innerHTML = '<i class="fas fa-play"></i>';
        isPlaying = false;
        
        audioProgress.style.width = '0%';
        currentTimeEl.textContent = '0:00';
    }

    function formatTime(seconds) {
        const mins = Math.floor(seconds / 60);
        const secs = Math.floor(seconds % 60);
        return `${mins}:${secs.toString().padStart(2, '0')}`;
    }

            totalDuration += (morse[i] === '.') ? dotDuration : dashDuration;
            if (i < morse.length - 1) totalDuration += spaceDuration;
        }
        
        const sampleRate = audioContext.sampleRate;
        const bufferLength = Math.ceil(totalDuration * sampleRate);
        const buffer = audioContext.createBuffer(1, bufferLength, sampleRate);
        const channelData = buffer.getChannelData(0);
        
        let time = 0;
        let sampleIndex = 0;
        
        for (let i = 0; i < morse.length; i++) {
            const char = morse[i];
            const duration = (char === '.') ? dotDuration : dashDuration;
            const frequency = 600;
            
            const toneSamples = Math.floor(duration * sampleRate);
            for (let j = 0; j < toneSamples; j++) {
                if (sampleIndex < bufferLength) {
                    channelData[sampleIndex] = Math.sin(2 * Math.PI * frequency * time) * 0.5;
                    sampleIndex++;
                    time += 1/sampleRate;
                }
            }
            
            if (i < morse.length - 1) {
                const spaceSamples = Math.floor(spaceDuration * sampleRate);
                for (let j = 0; j < spaceSamples; j++) {
                    if (sampleIndex < bufferLength) {
                        channelData[sampleIndex] = 0;
                        sampleIndex++;
                        time += 1/sampleRate;
                    }
                }
            }

        source.buffer = buffer;
        
        const gainNode = audioContext.createGain();
        gainNode.gain.value = 0.5;
        
        source.connect(gainNode);
        gainNode.connect(audioContext.destination);
        source.start(0);
    }

// ---------------- SIMPLE AUDIO DETECTION (STABLE DEMO VERSION) ----------------

const startAudioDetectionBtn = document.getElementById("start-audio-detection");
const audioDetectionOutput = document.getElementById("audio-detection-output");
const clearAudioDetectionBtn = document.getElementById("clear-audio-detection");

clearAudioDetectionBtn.style.display = "none";
let decodeTimeout = null;
let audioContextMic;
let analyser;
let microphone;
let stream;

let running = false;

let morse = "";
let lastState = false;
let lastChangeTime = 0;

// SAFE TIMINGS (works with your app)
const DOT_MAX = 200;
const LETTER_GAP = 300;
const WORD_GAP = 900;

function getVolume() {
    const data = new Uint8Array(analyser.fftSize);
    analyser.getByteTimeDomainData(data);

    let sum = 0;
    for (let i = 0; i < data.length; i++) {
        let v = (data[i] - 128) / 128;
        sum += v * v;
    }
    return Math.sqrt(sum / data.length) * 100;
}

let baseNoise = 0;

function calibrate() {
    let count = 0;
    let total = 0;

    function measure() {
        total += getVolume();
        count++;

        if (count < 50) {
            requestAnimationFrame(measure);
        } else {
            baseNoise = total / count;
            loop();
        }
    }

    measure();
}

function loop() {
    const now = Date.now();
    const volume = getVolume();

    const isSound = volume > baseNoise + 15;

    // state change
    if (isSound !== lastState) {

        const duration = now - lastChangeTime;

        if (lastState === true) {
    // sound ended

    if (duration < DOT_MAX) {
        morse += ".";
    } else {
        morse += "-";
    }

    // 🔥 AUTO DECODE after short delay
    if (decodeTimeout) clearTimeout(decodeTimeout);

    decodeTimeout = setTimeout(() => {
        decode();
    }, 500);


        } else {
            // silence ended

            if (duration > WORD_GAP) {
                decode();
                audioDetectionOutput.textContent += " ";
            } else if (duration > LETTER_GAP) {
    if (decodeTimeout) clearTimeout(decodeTimeout);
    decode();
}
        }

        lastState = isSound;
        lastChangeTime = now;
    }

    if (running) requestAnimationFrame(loop);
}

function decode() {
    if (morse.length === 0) return;

    const char = reverseMorseCode[morse] || "";
    audioDetectionOutput.textContent += char;

    if (audioDetectionOutput.textContent.trim().length > 0) {
        clearAudioDetectionBtn.style.display = "block";
    }

    morse = "";
}

// START / STOP
startAudioDetectionBtn.addEventListener("click", async () => {

    if (!running) {
        stream = await navigator.mediaDevices.getUserMedia({ audio: true });

        audioContextMic = new AudioContext();
        analyser = audioContextMic.createAnalyser();
        microphone = audioContextMic.createMediaStreamSource(stream);

        analyser.fftSize = 2048;
        microphone.connect(analyser);

        running = true;
        audioDetectionOutput.textContent = "Calibrating... stay silent";

        calibrate();

        startAudioDetectionBtn.innerHTML = '<i class="fas fa-stop"></i> Stop Audio Detection';

    } else {
        running = false;

        startAudioDetectionBtn.innerHTML = '<i class="fas fa-microphone"></i> Start Audio Detection';

        if (audioContextMic) audioContextMic.close();
        if (stream) stream.getTracks().forEach(t => t.stop());

        audioDetectionOutput.textContent += "\n(Audio stopped)";
    }
});

// CLEAR
clearAudioDetectionBtn.addEventListener("click", () => {
    audioDetectionOutput.textContent = "Waiting for Morse audio...";
    clearAudioDetectionBtn.style.display = "none";
});

const startLightBtn = document.getElementById("start-light-btn");
const lightOutput = document.getElementById("light-output");
const clearLightBtn = document.getElementById("clear-light-btn");

let lightRunning = false;
let video, canvas, ctx;

startLightBtn.addEventListener("click", async () => {

    if (!lightRunning) {

        // 🔴 STOP CAMERA DETECTION FIRST
        if (cameraActive) {
            cameraActive = false;
            clearInterval(cameraInterval);
            startCameraBtn.innerHTML = '<i class="fas fa-video"></i> Start Camera Detection';
            cameraOutput.textContent = "Camera stopped (switched to Light)";
        }

        video = document.getElementById("light-video");
        canvas = document.getElementById("light-canvas");
        ctx = canvas.getContext("2d");

        const stream = await navigator.mediaDevices.getUserMedia({ video: true });
        video.srcObject = stream;

        lightRunning = true;
        startLightBtn.innerHTML = "Stop Light Detection";

        detectLight();

    } else {
        lightRunning = false;
        startLightBtn.innerHTML = "Start Light Detection";

        if (video && video.srcObject) {
            video.srcObject.getTracks().forEach(t => t.stop());
        }
    }
});
// clear
clearLightBtn.addEventListener("click", () => {
    lightOutput.textContent = "Waiting for light input...";
    clearLightBtn.style.display = "none";
});
